import { useState } from "react";
import axios from "axios";
import Navbar from "./Navbar";
import FilterForm from "./FilterForm";
import ShowData from "./ShowData";
import UpdatePopup from "./UpdatePopup";

export default function Filter() {
    const [message, setMessage] = useState("");//for messages obtained from backend
    const [data, setData] = useState([]);//for getting the filtered data
    const [updateData, setUpdateData] = useState(null);//for getting the updated data
    const [isVisible, setIsVisible] = useState(false);
    const [refresh, setRefresh] = useState(false);


    const handleSubmit = async (formdata) => {

        try {
            // console.log(filterForm);
            const res = await axios.get("http://localhost:3000/api/expenses/filter", { params: { ...formdata } });
            if (res.data.success) {
                setData(res.data.data.expenses);
                setMessage(res.data.data.message);
            } else {
                setMessage("An error occured");
            }
        } catch (err) {
            setMessage("An error occurred");
        }
    }

    const handleDelete = async (id) => {
        try {
            await axios.delete(`http://localhost:3000/api/expenses/${id}`);
            setData(prev => prev.filter(item => item._id !== id));
        } catch (err) {
            console.error(err);
        }
    };

    const handleUpdate = async (row) => {
        setUpdateData(row);
        setIsVisible(true);
    }

    const handleUpdateSubmit = async (evt) => {
        evt.preventDefault(); // always first

        try {
            console.log(updateData);
            const res = await axios.put(
                `http://localhost:3000/api/expenses/${updateData._id}`,
                updateData
            );

            if (res.data.success) {
                setMessage(res.data.data.message);
                setIsVisible(false);
                setUpdateData(null);
                setRefresh(prev => !prev);
            } else {
                setMessage("An error occurred while updating");
            }

        } catch (err) {
            setMessage("An error occurred while updating");
            console.error(err);
        }
    };


    const onClose = async () => {
        setIsVisible(false);
    }

    return (
        <div>
            <Navbar setMessage={setMessage}></Navbar>
            <h1>Filter.jsx</h1>
            {message}
            <UpdatePopup isVisible={isVisible} updateData={updateData} setUpdateData={setUpdateData} handleUpdateSubmit={handleUpdateSubmit} onClose={onClose}></UpdatePopup>
            <FilterForm handleSubmit={handleSubmit}></FilterForm>
            <ShowData data={data} handleDelete={handleDelete} handleUpdate={handleUpdate}></ShowData>
        </div>
    )
}

                const [pieRes, barRes, lineRes] = await Promise.all([
                    axios.get("http://localhost:3000/api/analytics/category"),
                    axios.get("http://localhost:3000/api/analytics/paymentMode"),
                    axios.get("http://localhost:3000/api/analytics/month"),
                ]);

                setPieData(pieRes.data.data);
                setBarData(barRes.data.data);
                setLineData(lineRes.data.data);
            } catch (err) {
                console.error(err);
                setError("Failed to load analytics data");
            } finally {
                setLoading(false);
            }
        };

        fetchAnalytics();
    }, []);
